Create Dockerfile at the project root. The existing `juhayna/song-generation-levo:hf0613` image is incompatible with torch 2.6.0 + v2-large (per RESEARCH.md Pitfall 3 — image predates v2-large release). Build from scratch.

```dockerfile
# SongGeneration v2 RunPod Worker
# Base: CUDA 12.4 + cuDNN (matches torch==2.6.0 cu124 wheels)
FROM nvidia/cuda:12.4.1-cudnn-devel-ubuntu22.04

# Avoid interactive prompts during apt installs
ENV DEBIAN_FRONTEND=noninteractive

# ── System dependencies ────────────────────────────────────────────────────
RUN apt-get update && apt-get install -y --no-install-recommends \
    python3.11 \
    python3.11-dev \
    python3-pip \
    git \
    wget \
    ffmpeg \
    libsndfile1 \
    && rm -rf /var/lib/apt/lists/*

# Make python3.11 the default python
RUN update-alternatives --install /usr/bin/python python /usr/bin/python3.11 1 \
    && update-alternatives --install /usr/bin/python3 python3 /usr/bin/python3.11 1

# ── App directory ──────────────────────────────────────────────────────────
WORKDIR /app

# ── Clone SongGeneration repo (source code only — not pip-installable) ────
# The LeVoInference class lives in tools/gradio/levo_inference.py
# and must be importable from /app/SongGeneration on sys.path
RUN git clone https://github.com/tencent-ailab/SongGeneration.git /app/SongGeneration

# ── Install SongGeneration's own requirements ─────────────────────────────
# Install torch 2.6.0 first (cu124 wheel — NOT cu121; see Pitfall 4)
RUN pip install --no-cache-dir \
    torch==2.6.0 torchaudio==2.6.0 torchvision==0.21.0 \
    --index-url https://download.pytorch.org/whl/cu124

# Install remaining SongGeneration deps (excludes torch — already installed above)
RUN pip install --no-cache-dir \
    -r /app/SongGeneration/requirements.txt \
    --extra-index-url https://download.pytorch.org/whl/cu124

# ── Install worker-specific dependencies ──────────────────────────────────
COPY requirements.txt /app/requirements.txt
RUN pip install --no-cache-dir -r /app/requirements.txt \
    --extra-index-url https://download.pytorch.org/whl/cu124

# ── Download model weights at BUILD time (not at runtime) ─────────────────
# This ensures "Loading model" never appears during cold start wait time.
# Three directories required by LeVoInference (Pitfall 2: non-standard checkpoint structure):
#   /app/songgeneration_v2_large/  — model weights (config.yaml + model.pt)
#   /app/ckpt/                     — shared runtime checkpoint files
#   /app/third_party/              — Demucs separator and other runtime deps

# Download v2-large model weights
RUN pip install --no-cache-dir "huggingface_hub[cli]" && \
    huggingface-cli download lglg666/SongGeneration-v2-large \
        --local-dir /app/songgeneration_v2_large

# Download shared runtime checkpoints
RUN huggingface-cli download lglg666/SongGeneration-ckpt \
    --local-dir /app/ckpt

# Download third_party dependencies (Demucs separator model)
# NOTE: If third_party files are part of the SongGeneration repo (not HuggingFace),
# they may already be at /app/SongGeneration/third_party — verify after first build.
# If not present, additional download steps are needed here.

# ── Copy worker source code ────────────────────────────────────────────────
COPY handler.py generator.py storage.py schemas.py /app/

# ── Environment variable defaults (overridden at RunPod endpoint level) ───
ENV MODEL_CKPT_PATH=/app/songgeneration_v2_large
ENV SONGGEN_REPO_PATH=/app/SongGeneration
ENV USE_LOW_MEM=false
# SUPABASE_URL, SUPABASE_KEY, SUPABASE_BUCKET must be set at runtime

# ── Entrypoint ────────────────────────────────────────────────────────────
# handler.py imports generator.py which loads the model at module level.
# "Loading model" log line appears here — once per worker start.
CMD ["python", "-u", "handler.py"]
```

**docker-compose.yml** (project root):
```yaml
# Local development and GPU testing for the SongGeneration RunPod worker.
# Prerequisites:
#   1. NVIDIA Container Toolkit installed (nvidia-docker2)
#   2. .env file with SUPABASE_URL, SUPABASE_KEY, SUPABASE_BUCKET populated
#   3. docker build -t songgen-worker . (build the image first)
#
# Usage:
#   docker-compose up           — start worker with GPU
#   docker-compose run worker python -c "..."  — run ad-hoc command

version: "3.9"

services:
  worker:
    image: songgen-worker
    build:
      context: .
      dockerfile: Dockerfile
    env_file:
      - .env
    environment:
      # These override .env values if needed for local testing
      USE_LOW_MEM: "${USE_LOW_MEM:-false}"
      MODEL_CKPT_PATH: "/app/songgeneration_v2_large"
      SONGGEN_REPO_PATH: "/app/SongGeneration"
    deploy:
      resources:
        reservations:
          devices:
            - driver: nvidia
              count: 1
              capabilities: [gpu]
    volumes:
      # Mount local source for development iteration (without rebuilding image)
      - ./handler.py:/app/handler.py:ro
      - ./generator.py:/app/generator.py:ro
      - ./storage.py:/app/storage.py:ro
      - ./schemas.py:/app/schemas.py:ro
    stdin_open: true
    tty: true
```
