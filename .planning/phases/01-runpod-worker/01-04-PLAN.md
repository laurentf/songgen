```python
"""
RunPod serverless handler — entrypoint for the SongGeneration worker.

This module is intentionally thin. All business logic lives in generator.py.
The model is loaded at module level in generator.py; importing generate_batch
here triggers that load before runpod.serverless.start() is called, satisfying
the warm worker requirement (success criterion #3: "Loading model" once per worker).

RunPod job input shape:
  {"input": {"tracks": [{"idx": str, "gt_lyric": str, "descriptions": str|null}, ...]}}

RunPod job output shape:
  {"results": [{"idx": str, "status": "success"|"error", ...}]}
"""
import runpod
import structlog

# Importing generate_batch triggers MODEL loading in generator.py — do not move this import.
from generator import generate_batch

log = structlog.get_logger()


def handler(job: dict) -> dict:
    """RunPod serverless handler function.

    Called by RunPod for each job. Returns a JSON-serializable dict.
    All per-track errors are isolated inside generate_batch() — this
    function never raises an unhandled exception.
    """
    job_input = job.get("input", {})
    tracks = job_input.get("tracks", [])

    if not tracks:
        log.warning("No tracks in job input")
        return {"error": "No tracks provided", "results": []}

    log.info("Job received", track_count=len(tracks))
    results = generate_batch(tracks)
    log.info("Job complete", track_count=len(tracks), success=sum(1 for r in results if r.get("status") == "success"))

    return {"results": results}


# Model is already loaded. Start the serverless worker.
runpod.serverless.start({"handler": handler})
```
